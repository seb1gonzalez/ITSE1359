import sys
def sumEvens(n):
    count = 0
    sum = 0
    while count < n:
        sum += count
        count+=2
    print ("The sum of evens is ",sum)

def sumOdds(n):
    count = 0
    sum = 0
    while count < n:
        odd = 2*count+1
        sum += odd
        count+=1
    print ("The sum of odds is ",sum)

def oddAverage(n):
    count = 0
    sum = 0
    while count < n:
        odd = 2*count+1
        sum += odd
        count+=1
    avg = sum/n
    print ("The average of odds is ",avg)

def calcualteAvg(n):

        count+=1

def fib(n):

    if n == 1:
        return 1
    elif n == 0:
        return 0
    else:
        return fib(n-1) + fib(n-2)


def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n-1)
def addRange(n,m):
    count = n

    print("Addition in range is ",sum)
